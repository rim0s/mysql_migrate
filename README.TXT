#############################################################
# 使用流程
#############################################################
首次使用前：
# 1. 进入 RES 目录
cd RES

# 2. 运行部署脚本生成哈希文件
./deploy_validation_files.sh

# 3. 返回上级目录
cd ..

# 4. 运行主脚本
chmod +x mysql_migrate.sh
sudo ./mysql_migrate.sh yes
#############################################################
日常使用：
# 直接运行主脚本，它会自动加载验证系统
sudo ./mysql_migrate_V1.4.sh yes
#############################################################
验证文件完整性：
# 手动验证
cd RES
sha256sum -c mysql_validation_queries.sha256
#############################################################
关键注意事项 
   权限：主脚本和验证脚本都需要执行权限

    路径：保持目录结构不变

    哈希安全：.sha256 文件用于验证查询文件是否被修改

    兼容性：主脚本会检查所有必需文件是否存在
